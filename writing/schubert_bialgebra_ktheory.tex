\section{WC graphs}

\begin{definition}
A \emph{WC graph} is simply a finite subset $W\subseteq \mathbb{N}\times \mathbb{N}$ with no additional restrictions. We impose a total ordering on these pairs by declaring that $(a, b) \leq (c, d)$ if $a < c$ or $a = c$ and $b \geq d$. We define the \emph{weight} of a WC graph $W$ to be the sequence 
$$\wtt(W) = (i_1,\ldots,i_n)$$ 
where $i_j$ is the number of elements of $W$ in row $j$. We define the \emph{permutation} of a WC graph $W$ to be the permutation
$$\wof{W} = s(w_1)\bullet \cdots \bullet s(w_n)$$ 
where
$$w_i = (a_i, b_i)$$
is the $i$th element in the imposed ordering, and
$$s(a_i,b_i) = s_{a_i + b_i - 1}$$
with $\bullet$ being the Hecke product. The \emph{word} of $W$ is
$$(s(w_1),\ldots,s(w_n))$$

Define $\WC(w)$ to be the set of all WC graphs $W$ such that $\wof{W} = w$.
\end{definition}

The following is well-known.
\begin{theorem}
We have
$$\mathfrak{G}^{(\beta)}_w(x) = \sum_{W\in \WC(w)} \beta^{|W|-\ell(w)}x^{\wtt(W)}$$
\end{theorem}


\section{Grove polynomials and WC graphs}

\begin{definition}
A labeled indexed forest $(F,P)$ is an indexed forest $F$ with a function (labeling) $P:\mathrm{IN}(F)\to\mathbb{N}$ satisfying the following properties:
\begin{enumerate}
\item $P(\mathrm{left}(u)) < P(u)$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\leq P(\mathrm{right}(u))$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\geq \rho_F(u)$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\leq \lambda_F(u)$ for all $u\in \mathrm{IN}(F)$.
\end{enumerate}

The set $\mathrm{Komp}_P(F)$ is the set of functions $\kappa:\mathrm{IN}(F)\to 2^{\mathbb{N}}$ such that 

\begin{enumerate}
\item $\max(\kappa(u))\leq P(u)$ for all $u\in \mathrm{IN}(F)$.
\item $\max(\kappa(u))\leq \min(\kappa(\mathrm{left}(u)))$ for all $u\in \mathrm{IN}(F)$.
\item $\max(\kappa(u))< \min(\kappa(\mathrm{right}(u)))$ for all $u\in \mathrm{IN}(F)$.
\end{enumerate}
We have that
$$\wtt_i(\kappa) = \#\{u\in \mathrm{in}(F)\mid i\in \kappa(u)\}$$
\end{definition}


\begin{theorem}
Suppose a labeled indexed forest $(F,P)$ is such that there is a $\kappa\in\mathrm{Komp}_P(F)$ with 
$$\wtt(\kappa) = \code(F).$$ 
Then
$$\grove_F(x) = \sum_{\kappa\in\mathrm{Komp}_P(F)} \beta^{|\kappa|-|F|}x^{\wtt(\kappa)}$$
\end{theorem}



\begin{definition}
Let $W\in \WC(w)$ be a WC graph for $w$. Define via the Nadeau-Tewari insertion algorithm
$$(P(W), Q(W)) = (P(\mathrm{word}(W)), Q(\mathrm{word}(W)))$$
and let
$$F(W) = \mathrm{shape}(P(W))$$
Define $\code(F)$ to be the code of the indexed forest $F$.
\end{definition}


\begin{definition}
Let $(F,P)$ be a labeled indexed forest and let $\kappa\in\mathrm{Komp}_P(F)$. Consider the set 
$$A =\bigcup_{v\in\mathrm{IN}(F)}\kappa(v)\times \{P(v)\}$$
Define
$$\WC(\kappa) = \{(a_i, b_i + 1 - a_i)\mid (a_i, b_i)\in A\}$$
\end{definition}

\begin{theorem}
For a permutation $w$, let $\WC_\forest(w, F)$ be the set of $W\in\WC(w)$ such that $\code_\forest(W) = \code(F)$. Then we have that $\WC(w)$ is the disjoint union of $\WC(\mathrm{Komp}_{P(W)}(F(W)))$ for $W\in \WC_\forest(w, F)$.
% Furthermore, the Grothendieck polynomial $\mathfrak{G}^{(\beta)}_w(x)$ satisfies
% $$\mathfrak{G}^{(\beta)}_w(x)=\sum_{\substack{W\in \WC(w)\\\wtt(W) = \code(F(W))}} \beta^{|W|-\ell(w)}\grove_{F(W)}(x)$$
\end{theorem}

\begin{corollary}
The Grothendieck polynomial $\mathfrak{G}^{(\beta)}_w(x)$ satisfies
$$\mathfrak{G}^{(\beta)}_w(x)=\sum_{F\in \mathrm{IF}} \beta^{|F|-\ell(w)}a_w^F\grove_F(x)$$
where 
$$a_w^F = \#\{W\in \WC_\forest(w, F)\mid \wtt(W) = \code(F)\}$$
\end{corollary}

\begin{theorem}
Let $a$ be a composition. Let $w$ be the permutation such that $\code(w) = a$. Then we have that
$$\grove_a(x) = \sum_{W\in\WC_\forest(w, a)\cap \WC_{\mathrm{QY}}(w)} \mathcal{G}_{\wtt(W)}(x)$$
\end{theorem}

\begin{definition}
We define an invariant $\widetilde{P}_\forest(W)$ of a WC graph $W$ to be the WC graph $C$ such that $\code_\forest(C) = \wtt(C) = \code_\forest(W)$ and there exists a $\kappa\in \mathrm{Komp}_{P(C)}(F(C))$ such that $W\in \WC(\kappa)$.
\end{definition}

\begin{theorem}[Dual Grove LR rule]
Suppose
$$\grove_a^*\sqcupplus \grove_b^* = \sum_c P_{ab}^c\grove_c^*$$
for integers $P_{ab}^c$, with $\ell(a)=p$ and $\ell(b)=q$. Let $C$ be a WC graph with $\hr(C)=p+q$ such that $\code_\forest(C)=\wtt(C)=c$. Then $P_{ab}^c$ is the number of WC graphs $W$ with $\widetilde{P}_\forest(W) = \widetilde{P}_\forest(C)$ such that $\code_\forest(\clp^p(W)) = \wtt(\clp^p(W)) = a$ and $\code_\forest(\trm^p(W)) = \wtt(\trm^p(W)) = b$.
\end{theorem}

\section{Hecke insertion, \texorpdfstring{$K$}{K}-Knuth equivalence, and Lascoux classes}
\label{section:kknuth}

The purpose of this section is to attach to every WC graph a \emph{Lascoux
class}, namely the $K$-Knuth equivalence class of the increasing tableau obtained
by Hecke insertion of its word. The main input is a theorem, due to
Reiner--Yong~\cite{reiner2021grothlascoux} (conjecture) and
Shimozono--Yu~\cite{shimozono2021grothlascoux} (proof), stating that the WC
graphs of $w$ lying in a single $K$-Knuth class have generating function equal to
a single Lascoux polynomial. This refines the symmetric statement of
Buch--Kresch--Shimozono--Tamvakis--Yong~\cite{buch2008factorsequences}. We phrase
everything in the WC-graph language of the previous sections, so that it can be
used to show that WC-graph transition, and hence the squash product, preserves
Lascoux classes.

Let $\partial_i$ denote the $i$th divided difference operator,
$$\partial_i f=\frac{f-s_i f}{x_i-x_{i+1}},$$
and define the \emph{$K$-theoretic isobaric (Demazure) operator}
$$\pib_i f=\partial_i\bigl(x_i(1+\beta x_{i+1})f\bigr).$$
Then $\pib_i(1)=1$, $\pib_i$ sends polynomials to polynomials whose value is
symmetric in $x_i,x_{i+1}$, and at $\beta=0$ it recovers the Demazure operator
$\pi_i f=\partial_i(x_i f)$ of the key polynomials.

\begin{definition}
\label{definition:lascoux-poly}
For a weak composition $\alpha=(\alpha_1,\ldots,\alpha_n)\in\mathbb{N}^n$ the
\emph{Lascoux polynomial} $\lasc_\alpha$ is defined by:
\begin{enumerate}
\item if $\alpha_1\geq\alpha_2\geq\cdots\geq\alpha_n$ (i.e. $\alpha$ is a
partition), then $\lasc_\alpha=x^\alpha=\prod_i x_i^{\alpha_i}$;
\item if $\alpha_i<\alpha_{i+1}$ for some $i$, then
$\lasc_\alpha=\pib_i\,\lasc_{s_i\alpha}$, where $s_i\alpha$ interchanges
$\alpha_i$ and $\alpha_{i+1}$.
\end{enumerate}
This is well defined by~\cite{shimozono2021grothlascoux}. The lowest-$\beta$
component of $\lasc_\alpha$ (its terms of total degree $|\alpha|$, i.e. the
$\beta^0$ part) is the key (Demazure) polynomial $\kappa_\alpha$.
\end{definition}

\subsection{The Grothendieck-to-Lascoux expansion}

Recall from the definition of a WC graph that the elements of $W$ are listed as
$w_1<w_2<\cdots<w_n$ in the total order on $\mathbb{N}\times\mathbb{N}$, with
$w_i=(a_i,b_i)$, and that
$$s(a_i,b_i)=s_{a_i+b_i-1}.$$

\newcommand{\hi}{\mathfrak{L}}
\begin{definition}
For a WC graph $W$, define the \emph{diagonal skew tableau} $\dwd(W)$ to be the unique tableau of skew shape $\mathrm{st}(|W|)/\mathrm{st}(|W|-1)$ whose reading word is $\mathrm{word}(W)$. The Lascoux invariant $\hi(W)$ of $W$ is the greedy rectification of $\dwd(W)$. We can in fact apply this to any word.
\end{definition}

\begin{theorem}[Dual Lascoux LR rule]
Suppose
$$\mathcal{L}_a^*\sqcupplus \mathcal{L}_b^* = \sum_c L_{ab}^c\mathcal{L}_c^*$$
for integers $L_{ab}^c$, with $\ell(a)=p$ and $\ell(b)=q$. Let $C$ be a WC graph with $\hr(C)=p+q$ such that $\wtt(\hi(C)) = \wtt(C) = c$. Then $L_{ab}^c$ is the number of WC graphs $W$ with $\hi(W) = \hi(C)$ such that $\wtt(\hi(\clp^p(W))) = \wtt(\clp^p(W)) = a$ and $\wtt(\hi(\trm^p(W))) = \wtt(\trm^p(W)) = b$.
\end{theorem}


The next theorem is the symmetric statement of
Buch--Kresch--Shimozono--Tamvakis--Yong, in the WC-graph normalization: grouping
the WC graphs of $w$ by Lascoux class collects their generating function into
symmetric Grothendieck building blocks.

\begin{theorem}[\cite{buch2008factorsequences}]
\label{theorem:bksty}
Fix an increasing tableau $P$ of straight shape $\lambda$ whose row reading word
has Hecke product $w$. Then
$$\sum_{\substack{W\in\WC(w)\\ \Pw(W)=P}}\beta^{|W|-\ell(w)}x^{\wtt(W)}$$
is symmetric in the variables it involves; summed over all such $P$ of a fixed
shape $\lambda$ it equals the stable (Grassmannian) Grothendieck contribution
$a_{w,\lambda}\,G_\lambda$, recovering the factor-sequence expansion
$G_w=\sum_\lambda a_{w,\lambda}G_\lambda$.
\end{theorem}

The nonsymmetric refinement replaces each symmetric block by a single Lascoux
polynomial. This is the ``Grothendieck to Lascoux'' expansion.

\begin{theorem}[Reiner--Yong~\cite{reiner2021grothlascoux};
Shimozono--Yu~\cite{shimozono2021grothlascoux}]
\label{theorem:groth-lascoux}
There is a map $P\mapsto\alpha(P)$ assigning to each increasing tableau $P$ a
weak composition $\alpha(P)$, with underlying partition $\mathrm{sh}(P)$,
determined by the $K$-theoretic right key of $P$, such that for every increasing
tableau $P$ arising as an insertion tableau of some $W\in\WC(w)$,
$$\sum_{\substack{W\in\WC(w)\\ \Pw(W)=P}}\beta^{|W|-\ell(w)}x^{\wtt(W)}
   =\lasc_{\alpha(P)}.$$
Consequently
$$\mathfrak{G}^{(\beta)}_w(x)=\sum_{P\in\mathcal{I}(w)}\lasc_{\alpha(P)}
   =\sum_{\alpha}c_{w,\alpha}\,\lasc_{\alpha},\qquad
   c_{w,\alpha}=\#\{P\in\mathcal{I}(w)\mid\alpha(P)=\alpha\}\in\mathbb{Z}_{\geq0},$$
where $\mathcal{I}(w)=\{\Pw(W)\mid W\in\WC(w)\}$ is the set of Lascoux classes of
$w$.
\end{theorem}

The content we use below is the following: \emph{the Lascoux polynomial attached
to a WC graph depends only on its Lascoux class.} We isolate this as a corollary.

\begin{corollary}
\label{corollary:class-determines-lascoux}
The assignment
$$W\longmapsto \lasc_{\alpha(\Pw(W))}$$
factors through the Lascoux class $[W]=\Pw(W)$. In particular, if $\Phi$ is any
map on WC graphs with $\Pw(\Phi(W))\kk\Pw(W)$ for all $W$, equivalently
$\dwd(\Phi(W))\kk\dwd(W)$, then
$$\lasc_{\alpha(\Pw(\Phi(W)))}=\lasc_{\alpha(\Pw(W))}$$
for all $W$; that is, $\Phi$ preserves Lascoux classes and hence the associated
Lascoux polynomial.
\end{corollary}

\begin{proof}
Immediate from Theorem~\ref{theorem:kknuth-invariant}(1), which identifies
$\Pw(W)$ with the $K$-Knuth class of $\dwd(W)$, and
Theorem~\ref{theorem:groth-lascoux}, in which $\alpha$ is a function of $\Pw(W)$
alone.
\end{proof}

\begin{remark}
Corollary~\ref{corollary:class-determines-lascoux} is the $K$-theoretic,
straight-shape analogue of the grove/forest decomposition of
Section~\ref{section:kknuth}'s predecessor: there the WC graphs of $w$ are
grouped by the indexed forest $F(W)$ of the Nadeau--Tewari insertion and each
fiber contributes $\grove_{F(W)}$, whereas here they are grouped by the
$K$-Knuth class $\Pw(W)$ of Hecke insertion and each fiber contributes
$\lasc_{\alpha(\Pw(W))}$. Identifying the two groupings, e.g. establishing an
equality $\grove_{F}=\lasc_{\alpha(\Pw)}$ under the correspondence
$F\leftrightarrow\Pw$, would identify grove polynomials with Lascoux
polynomials.
\end{remark}

\section{Bounded WC graphs and the \texorpdfstring{$K$}{K}-theoretic transition}
\label{section:kzero}

We now equip WC graphs with a row grading and a transition map, mirroring the
treatment of bounded RC graphs in the reduced theory of \cite{samuelmolev}.
Everything in this section specializes to those reduced constructions on the
subset of RC graphs.

\begin{definition}\label{definition:bwc}
A \emph{bounded WC graph} is a WC graph $W$ together with a specified number of
rows $n\geq \max\{a: (a,b)\in W\text{ for some }b\}$ subject to
$\maxd(\wof{W})\leq n$. We write $(W,n)$ for the pair, abuse notation to treat
the number of rows as a property of $W$, and call $\hr(W)=n$ the \emph{height}
of $W$. For $\hr(W)\geq 1$ we define the \emph{trim} $\trm(W)$ to be the bounded
WC graph with height $\hr(W)-1$ and underlying set
$$\trm(W)=\{(a-1,b-1): (a,b)\in W,\ a\geq 2\},$$
and we set $\shup^m W=\{(a+m,b)\mid (a,b)\in W\}$; these are the bounded RC-graph
operations of \cite{samuelmolev} applied to WC graphs verbatim. Let $\BWC$ be
the free abelian group on all bounded WC graphs, graded by height,
$$\BWC=\bigoplus_{n=0}^{\infty}\BWC_n.$$
\end{definition}

A WC graph with $\ell(\wof{W})=|W|$ is exactly an RC graph, and on such graphs
$\trm$ and $\shup^m$ agree with the RC operations. Thus $\BRC\subseteq\BWC$ is a
graded subgroup and the projection $\rho:\BWC\to\BRC$ that kills every WC graph
with $|W|>\ell(\wof{W})$ is a graded retraction.

\subsection{The transition map via marked bumpless pipe dreams}

In the reduced case the map $\zeromap$ of \cite{samuelmolev} realizes the
Lascoux--Sch\"utzenberger transition on RC graphs. Its $K$-theoretic counterpart
is most naturally described through bumpless pipe dreams. Recall the following
ingredients.

\begin{enumerate}
\item \textbf{Marked bumpless pipe dreams.} Weigandt \cite{Weigandt2021}
introduced \emph{marked bumpless pipe dreams} (marked BPDs) and proved that the
double Grothendieck polynomial $\mathfrak{G}^{(\beta)}_w$ is the generating
function of marked BPDs of $w$, extending the Lam--Lee--Shimozono formula from
$K$-theory. Let $\mathrm{MBPD}(w)$ denote the set of marked BPDs of $w$.

\item \textbf{The HSY bijection.} Huang, Shimozono, and Yu
\cite{HuangShimozonoYu2024} construct a weight-preserving bijection between
marked BPDs and reverse compatible pairs, the latter being in bijection with
not-necessarily-reduced pipe dreams; their bijection generalizes, in the
unmarked reduced case, the canonical bijection of Gao and Huang
\cite{GaoHuang2023}. Reading a bounded WC graph as the compatible sequence of a
(not necessarily reduced) pipe dream, this yields a weight-preserving bijection
$$\Psi:\;\bigcup_{n}\{W\in\BWC_n:\wof{W}=w\}\;\longrightarrow\;\mathrm{MBPD}(w),$$
whose restriction to RC graphs is the Gao--Huang bijection $\phi$ used in the
reduced theory of \cite{samuelmolev}.

\item \textbf{Weigandt transition.} Weigandt's droop/transition operation
$\mathbf{t}$ on bumpless pipe dreams removes the bottom row and rearranges the
resulting diagram into a valid BPD of one fewer row, realizing the
(co)transition recursion for Grothendieck polynomials; it acts on marked BPDs
with an empty bottom row \cite{Weigandt2021,HuangShimozonoYu2024}.
\end{enumerate}

\begin{definition}\label{definition:lzeromap}
Let $W$ be a bounded WC graph with $\hr(W)=n$ whose row $n$ is empty. Define the
\emph{$K$-theoretic transition map}
$$\lzeromap(W)=\Psi^{-1}\bigl(\mathbf{t}(\Psi(W))\bigr),$$
a bounded WC graph of height $n-1$. We extend $\lzeromap$ to an endomorphism of
$\BWC$ by setting $\lzeromap(W)=0$ when row $n$ is nonempty.
\end{definition}

The following properties are the $K$-theoretic analogues of the reduced-case
transition lemma, the trim-commutation lemma, and the zeroing bijection of
\cite{samuelmolev}. We record them as the working hypotheses of the
construction; each specializes to its reduced counterpart under $\rho$, where it
is a theorem of \cite{samuelmolev}.

\begin{proposition}\label{proposition:lzero-reduces}
On RC graphs the map $\lzeromap$ agrees with $\zeromap$: for every bounded WC
graph $W$ with empty last row,
$$\rho(\lzeromap(W))=\zeromap(\rho(W)),$$
and in particular $\lzeromap|_{\BRC}=\zeromap$.
\end{proposition}
\begin{proof}[Proof sketch]
On RC graphs $\Psi$ is the Gao--Huang bijection $\phi$ and $\mathbf{t}$
restricts to Weigandt's transition on reduced BPDs, so
$\lzeromap=\phi^{-1}\circ\mathbf{t}\circ\phi$ there. The reduced-case identity
$\zeromap\circ\phi^{-1}=\phi^{-1}\circ\mathbf{t}$
(the conjecture of \cite{samuelmolev} relating $\zeromap$ and $\mathbf{t}$
through $\phi$) gives the claim; $\rho$ discards the non-reduced terms, none of
which are produced from a reduced input.
\end{proof}

\begin{proposition}\label{proposition:lzero-transition}
Let $W$ be a bounded WC graph with $\hr(W)=n\geq 1$ and empty row $n$. Then
$\hr(\lzeromap(W))=n-1$, the weight is preserved,
$$\wtt(\lzeromap(W))=\wtt(W),$$
and the permutation undergoes the $K$-theoretic (Grothendieck) transition:
$\wof{\lzeromap(W)}$ is obtained from $\wof{W}$ by a single step of the
Grothendieck cotransition recursion in row $n$.
\end{proposition}

\begin{proposition}\label{proposition:lztrim}
Let $(W,n)$ be a bounded WC graph with $n\geq 2$ whose row $n$ is empty. Then
$$\lzeromap(\trm(W))=\trm(\lzeromap(W)).$$
\end{proposition}
\begin{proof}[Proof sketch]
This is the $K$-analogue of the trim-commutation lemma of \cite{samuelmolev}.
Trimming preserves a suffix of the word of $W$, and both $\Psi$ and $\mathbf{t}$
act on the bottom rows of the marked BPD without reference to the top row: the
transition $\mathbf{t}$ proceeds downward from the removed bottom row, so
discarding the top row commutes with it. Under $\rho$ the identity specializes
to the reduced statement of \cite{samuelmolev}.
\end{proof}

\begin{proposition}\label{proposition:lzero-bijection}
Let $w$ be a permutation with last descent at most $n$, and let
$\WC(w,n)^0$ be the set of bounded WC graphs $W$ with $\hr(W)=n$, $\wof{W}=w$,
and empty row $n$. Then
$$\lzeromap:\;\WC(w,n)^{0}\;\longrightarrow\;\bigcup_{w\downvar{n} w'}\WC(w',n-1)$$
is a weight-preserving bijection, where $w'$ ranges over the outputs of the
Grothendieck cotransition in row $n$. Consequently
$$\lzeromap\Bigl(\sum_{\substack{\wof{W}=w\\\hr(W)=n}}W\Bigr)=\sum_{w\downvar{n} w'}\ \sum_{\substack{\wof{W'}=w'\\\hr(W')=n-1}}W',$$
the WC-graph form of the Grothendieck transition formula.
\end{proposition}

\section{WC Graph ring}

We now lift the $K$-theoretic transition map $\lzeromap$ to an associative
product on $\BWC$, in exact parallel with the product $\sqcupplus$ on $\BRC$
built from $\zeromap$ in \cite{samuelmolev}. Only the underlying zeroing map has
changed: $\trm^p$ is unchanged, and $\clp^p$ is now built from $\lzeromap$.

\begin{definition}\label{definition:kcliptrim}
Let $W$ be a bounded WC graph with $\hr(W)=n$ and let $0\leq p\leq n$. Define
$\trm^p(W)\in\BWC_{n-p}$ to be the result of discarding the top $p$ rows of $W$
and reindexing (equivalently, the $p$-fold iterate of $\trm$), and define
$\clp^p(W)\in\BWC_{p}$ to be the $(n-p)$-fold iterate of $\lzeromap$ applied to
the bottom row,
$$\clp^p(W)=\lzeromap^{\,n-p}(W),$$
which is defined whenever each intermediate bottom row is empty at the moment it
is zeroed (and is $0$ otherwise). Thus $\clp^p(W)$ and $\trm^p(W)$ decompose $W$
across the horizontal cut between rows $p$ and $p+1$.
\end{definition}

Associativity of the product below rests on two structural identities relating
$\clp^p$ and $\trm^p$: functoriality of $\clp$ under iteration, and commutativity
of $\clp$ with $\trm$. The first is formal; the second is the content of
Proposition~\ref{proposition:lztrim}.

\begin{lemma}\label{lemma:kclpfunc}
Let $W$ be a bounded WC graph with $\hr(W)=n$ and $0\leq p\leq p+q\leq n$. Then
$$\clp^p(\clp^{p+q}(W))=\clp^p(W).$$
\end{lemma}
\begin{proof}
Both sides are computed by iterating $\lzeromap$ on the bottom row. With
$r=n-(p+q)$, the graph $\clp^{p+q}(W)$ is the $r$-fold iterate $\lzeromap^{r}(W)$
and $\clp^p(W)=\lzeromap^{r+q}(W)$; the first $r$ iterations in the computation
of $\clp^p(W)$ produce $\clp^{p+q}(W)$, and the remaining $q$ produce
$\clp^p(\clp^{p+q}(W))$. Hence the two agree.
\end{proof}

\begin{lemma}[Commutativity of $\clp$ and $\trm$]\label{lemma:kclptrm}
Let $W$ be a bounded WC graph with $\hr(W)=p+q+r$. Then
$$\trm^p(\clp^{p+q}(W))=\clp^q(\trm^p(W)).$$
\end{lemma}
\begin{proof}
Both sides have height $q$. By Proposition~\ref{proposition:lztrim},
$\lzeromap(\trm(W'))=\trm(\lzeromap(W'))$ whenever the last row of $W'$ is empty.
Since $\clp^{p+q}$ on a graph of height $p+q+r$ and $\clp^q$ on a graph of height
$q+r$ are both the $r$-fold iterate of $\lzeromap$ on the bottom row
(cf.\ Lemma~\ref{lemma:kclpfunc}), iterating
Proposition~\ref{proposition:lztrim} $r$ times gives
$$\trm^p(\clp^{p+q}(W))=\trm^p(\lzeromap^{r}(W))=\lzeromap^{r}(\trm^p(W))=\clp^q(\trm^p(W)),$$
as required.
\end{proof}

\begin{definition}\label{definition:kringproduct}
For bounded WC graphs $W_1$ with $\hr(W_1)=m$ and $W_2$ with $\hr(W_2)=n$, define
$$W_1\sqcupplus W_2=\sum_{\substack{\clp^m(W')=W_1\\\trm^m(W')=W_2}}W',$$
where $W'$ ranges over bounded WC graphs with $\hr(W')=m+n$, and extend
bilinearly to $\BWC$.
\end{definition}

\begin{theorem}\label{theorem:kring}
The product $\sqcupplus$ is associative, and $\BWC$ is a ring under $\sqcupplus$
with identity the empty WC graph of height $0$.
\end{theorem}
\begin{proof}
Let $W_1,W_2,W_3$ be bounded WC graphs of heights $p,q,r$, and let $W$ be a
bounded WC graph with $\hr(W)=p+q+r$. Unwinding
Definition~\ref{definition:kringproduct} in each bracketing, $W$ contributes to
the doubly-bracketed products under the conditions
\begin{center}
\begin{tabular}{l|l|l}
& Left $(W_1\sqcupplus W_2)\sqcupplus W_3$ & Right $W_1\sqcupplus(W_2\sqcupplus W_3)$ \\
\hline
(i) & $\clp^p(\clp^{p+q}(W))=W_1$ & $\clp^p(W)=W_1$ \\
(ii) & $\trm^p(\clp^{p+q}(W))=W_2$ & $\clp^q(\trm^p(W))=W_2$ \\
(iii) & $\trm^{p+q}(W)=W_3$ & $\trm^q(\trm^p(W))=W_3$ \\
\end{tabular}
\end{center}
Row (i) is equivalent on both sides by Lemma~\ref{lemma:kclpfunc}, row (ii) by
Lemma~\ref{lemma:kclptrm}, and row (iii) because $\trm^{p+q}=\trm^q\circ\trm^p$
by definition of $\trm$. Each condition is an equality carrying no combinatorial
multiplicity, so $W$ occurs with multiplicity $1$ in either bracketing, and the
two bracketed products agree. Distributivity is imposed by bilinear extension,
and the empty height-$0$ WC graph is a two-sided identity.
\end{proof}

Because $\lzeromap$ restricts to $\zeromap$ on RC graphs
(Proposition~\ref{proposition:lzero-reduces}) and $\trm$ is unchanged, the
retraction $\rho$ intertwines the two clip/trim decompositions; hence it is
compatible with the products.

\begin{proposition}
The linear map
$$\rho:\BWC\to\BRC$$
such that
$$\rho(W) = \begin{cases}
W&\text{if $W$ is an RC graph}\\
0&\text{otherwise}
\end{cases}$$
is a surjective ring homomorphism.
\end{proposition}

\section{The \texorpdfstring{$K$}{K}-plactic monoid and algebra}
\label{section:kplactic}

The reduced theory of \cite{samuelmolev} attaches to each rank $k$ a
\emph{plactic monoid} $\plac_k$: the set of height-$k$ RC graphs whose
permutation is $k$-Grassmannian, made into a monoid by the squash product, which
is a combinatorial model for the ring of Schur polynomials in $k$ variables. We
now introduce its $K$-theoretic, set-valued analogue. It is obtained by replacing
RC graphs with WC graphs and, correspondingly, semistandard Young tableaux and
Schur polynomials with their set-valued and stable Grothendieck counterparts.

Recall that a permutation $w$ is $k$-\emph{Grassmannian} if it is either the
identity or has a unique descent, located at position $k$. For such a $w$ the
associated partition
$$\lambda(w)=\bigl(w(k)-k,\ w(k-1)-(k-1),\ \ldots,\ w(1)-1\bigr)$$
has at most $k$ parts, and the Grothendieck polynomial
$\mathfrak{G}^{(\beta)}_w$, expressed in the variables $x_1,\ldots,x_k$, is the
stable (Grassmannian) Grothendieck polynomial $G_{\lambda(w)}(x_1,\ldots,x_k)$;
its lowest-degree component is the Schur polynomial
$s_{\lambda(w)}(x_1,\ldots,x_k)$. By the WC-graph formula
$\mathfrak{G}^{(\beta)}_w(x)=\sum_{W\in\WC(w)}\beta^{|W|-\ell(w)}x^{\wtt(W)}$,
the height-$k$ WC graphs of a $k$-Grassmannian permutation are the set-valued
analogue of the semistandard tableaux enumerated by a Schur polynomial.

\begin{definition}[Squash product]\label{definition:ksquash}
Let $W_1$ and $W_2$ be bounded WC graphs of height $k$. Their \emph{disjoint
union} $W_1\star W_2$ is formed by shifting every entry of $W_2$ upward far
enough that each reflection $s(a,b)$ used by $W_2$ exceeds---and hence Hecke
commutes with---every reflection used by $W_1$, and then overlaying the two
graphs row by row. The weight is additive, $\wtt(W_1\star
W_2)=\wtt(W_1)+\wtt(W_2)$. The \emph{squash product} is
$$W_1\boxtimes W_2=\clp^{k}(W_1\star W_2),$$
the clip of the disjoint union back down to $k$ rows
(Definition~\ref{definition:kcliptrim}).
\end{definition}

On $k$-Grassmannian factors the squash product is exactly the single surviving
term of the ring product $\sqcupplus$ of
Definition~\ref{definition:kringproduct}: since the clip $\clp^{k}$ and trim
$\trm^{k}$ of $W_1\star W_2$ recover $W_1$ and $W_2$ respectively,
$$W_1\boxtimes W_2 = W_1\sqcupplus W_2\qquad(\hr(W_1)=\hr(W_2)=k).$$
In particular $\boxtimes$ is associative, being the restriction of the
associative product $\sqcupplus$ (Theorem~\ref{theorem:kring}) to
$n$-Grassmannian WC graphs.

\begin{definition}[$K$-plactic monoid and algebra]\label{definition:kplac-monoid}
The \emph{$K$-plactic monoid of rank $k$} is the set
$$\kplac_k=\bigl\{\,W\in\BWC_k:\wof{W}\text{ is }k\text{-Grassmannian}\,\bigr\}$$
of height-$k$ WC graphs whose permutation has no descent or a single descent at
$k$, equipped with the squash product $\boxtimes$. It is a monoid with unit the
empty height-$k$ WC graph. The \emph{$K$-plactic algebra} $\mathbb{Z}\kplac_k$ is
its monoid algebra: the free abelian group on $\kplac_k$ with the bilinear
extension of $\boxtimes$.
\end{definition}

\begin{theorem}\label{theorem:kplac-monoid}
The set $\kplac_k$ is closed under $\boxtimes$, and $(\kplac_k,\boxtimes)$ is an
associative monoid. It is the set-valued analogue of the plactic monoid
$\plac_k$ of semistandard tableaux of at most $k$ rows: the degeneration
that discards every WC graph $W$ with $|W|>\ell(\wof{W})$ (equivalently, restricts
to RC graphs) sends $\kplac_k$ onto $\plac_k$ and $\boxtimes$ to the classical
squash product.
\end{theorem}
\begin{proof}
Closure holds because $\clp^{k}$ carries a height-$2k$ WC graph whose top and
bottom halves are $k$-Grassmannian to a height-$k$ WC graph whose permutation is
again $k$-Grassmannian: the transition $\lzeromap$
(Proposition~\ref{proposition:lzero-transition}) preserves the property of having
a single descent at the top row. Associativity is the restriction of
Theorem~\ref{theorem:kring}. The degeneration is the restriction of the
retraction $\rho:\BWC\to\BRC$ (which is a ring homomorphism), and on RC graphs
$\clp^{k}=\zeromap^{\,k}$ agrees with the reduced clip of \cite{samuelmolev}, so
$\rho$ intertwines $\boxtimes$ with the reduced squash product.
\end{proof}

The squash product realizes stable Grothendieck multiplication, in exact
parallel with the crystal isomorphism of the reduced theory
\cite{samuelmolev}. We record this as the $K$-theoretic analogue.

\begin{theorem}\label{theorem:squash-crystal-wc}
Let $u,v$ be $k$-Grassmannian permutations. Then the disjoint union induces a
weight-preserving bijection
$$\{W\in\kplac_k:\wof{W}=u\}\times\{W'\in\kplac_k:\wof{W'}=v\}
\ \longrightarrow\ \{U\in\BWC_{2k}:\clp^{k}(U)\in\kplac_k(u),\ \trm^{k}(U)\in\kplac_k(v)\},$$
and clipping back to $k$ rows decomposes the result according to the
$K$-theoretic Littlewood--Richardson (Buch) rule
\cite{buch2008factorsequences}: writing
$\lambda,\mu$ for the partitions of $u,v$,
$$\sum_{\substack{W\in\kplac_k(u)\\ W'\in\kplac_k(v)}}W\boxtimes W'
=\sum_{\nu}c^{\nu}_{\lambda\mu}(\beta)\!\!\sum_{\substack{W''\in\kplac_k\\ \lambda(\wof{W''})=\nu}}\!\!W'',$$
where $c^{\nu}_{\lambda\mu}(\beta)$ are the structure constants of the stable
Grothendieck polynomials, $G_\lambda G_\mu=\sum_\nu c^{\nu}_{\lambda\mu}(\beta)G_\nu$.
\end{theorem}

For a $k$-Grassmannian permutation $w$ we assemble the WC graphs of $w$ into a
single element of the $K$-plactic algebra.

\begin{definition}\label{definition:groth-element}
For a $k$-Grassmannian permutation $w$, the \emph{Grothendieck element} is
$$\mathbf{G}_w=\sum_{\substack{W\in\kplac_k\\ \wof{W}=w}}W\ \in\ \mathbb{Z}\kplac_k,$$
the sum of all height-$k$ WC graphs with permutation $w$. Under the weight map
$W\mapsto\beta^{|W|-\ell(w)}x^{\wtt(W)}$ it maps to the stable Grothendieck
polynomial $G_{\lambda(w)}(x_1,\ldots,x_k)$.
\end{definition}

\begin{theorem}\label{theorem:groth-subring}
As $w$ ranges over the $k$-Grassmannian permutations, the Grothendieck elements
$\mathbf{G}_w$ are linearly independent and span a commutative subring
$\mathbf{G}_k\subseteq\mathbb{Z}\kplac_k$. The weight map induces a ring
isomorphism
$$\mathbf{G}_k\ \xrightarrow{\ \sim\ }\ \mathbb{Z}\bigl[G_{\lambda}(x_1,\ldots,x_k):\lambda\ \text{a partition with}\le k\ \text{parts}\bigr],\qquad
\mathbf{G}_w\longmapsto G_{\lambda(w)}(x_1,\ldots,x_k),$$
onto the ring generated by the stable Grothendieck polynomials in $k$ variables.
Passing to lowest degree recovers the classical model: $\mathbf{G}_k$ degenerates
to the ring of symmetric polynomials spanned by the Schur polynomials
$s_\lambda(x_1,\ldots,x_k)$, and $\mathbf{G}_w$ to the Schur element of
$\plac_k$.
\end{theorem}
\begin{proof}
Distinct $k$-Grassmannian permutations have disjoint sets of WC graphs, so the
$\mathbf{G}_w$ are linearly independent. By
Theorem~\ref{theorem:squash-crystal-wc} the squash product of two Grothendieck
elements decomposes with the same structure constants as the product of the
corresponding stable Grothendieck polynomials,
$$\mathbf{G}_u\boxtimes\mathbf{G}_v=\sum_{w}c^{w}_{u,v}(\beta)\,\mathbf{G}_w,
\qquad
G_{\lambda(u)}\,G_{\lambda(v)}=\sum_{w}c^{w}_{u,v}(\beta)\,G_{\lambda(w)},$$
where $c^{w}_{u,v}(\beta)$ are the $K$-theoretic Littlewood--Richardson
(Buch) coefficients. Hence the span $\mathbf{G}_k$ is a subring, the assignment
$\mathbf{G}_w\mapsto G_{\lambda(w)}$ is a ring homomorphism, and it is bijective
onto the stated ring because the stable Grothendieck polynomials of partitions
with at most $k$ parts are a $\mathbb{Z}$-basis of it. Commutativity follows since
the target ring is commutative. Setting $\beta=0$ (equivalently, retaining the
lowest-degree part, which is $\rho$ followed by symmetrization) sends
$G_\lambda\mapsto s_\lambda$ and recovers the plactic model of \cite{samuelmolev}.
\end{proof}

\begin{remark}
Theorem~\ref{theorem:groth-subring} is the $K$-theoretic analogue of the
statement that, inside the plactic algebra, the sums of tableaux of fixed shape
span a commutative ring isomorphic to the ring of Schur polynomials. Here shapes
are replaced by $k$-Grassmannian permutations, tableaux by WC graphs,
Knuth equivalence by $K$-Knuth equivalence (Section~\ref{section:kknuth}), and
Schur polynomials by stable Grothendieck polynomials.
\end{remark}

algebras and assemble into a tensor model for $\BWC_n$.

For each $n>0$ there is a surjective ring homomorphism
$$\mathbb{Z}\kplac_n\to \mathbb{Z}\plac_n$$
obtained on elementary symmetric WC graphs by sending $E_{p,k}^{\beta}$ with
$|\beta|>p$ to $E_{|\beta|,k}^{\beta}$; it is the restriction of the retraction
$\rho$ and provides a section of the inclusion $\plac_n\hookrightarrow\kplac_n$.
Define
$$\mathcal{M}_n^K = \bigotimes_{i=1}^n \mathbb{Z}\kplac_i.$$
There is an evident homomorphism $\mathcal{M}_n^K\to \mathcal{M}_n$ obtained by
applying the above homomorphism to each factor, giving a commutative diagram
$$\begin{tikzcd}
\mathcal{M}_n^K \arrow[r]\arrow[d]  & \mathcal{M}_n\arrow[d]  \\
\BWC_n \arrow[r, dashed] & \BRC_n
\end{tikzcd}$$
\end{definition}

\emph{Claim}: Every Grothendieck polynomial $\mathfrak{G}_w(x)$ can be expressed as a sum of products of ordinary elementary symmetric stable Grothendieck polynomials with integer coefficients, with the number of factors of the product fixed (allowing degree $0$ factors) each being in a fixed number of $x$ variables.

Let $\mathfrak{K}^i$ denote the operator $(1 + y_{i + 1})\partial^i - 1$, acting on the $y$ variables. Consider the product of operators
$$\mathfrak{K}^{i + p}\cdots \mathfrak{K}^{i + 1}\mathfrak{K}^{i} = ((1+ y_{i+p})\partial^{i+p} - 1)\cdots ((1 + y_{i+1})\partial^{i+1} - 1)((1 + y_i)\partial^i - 1)E_p(\tau^i y;1)$$
We have that the top degree factorial elementary stable Grothendieck polynomial in $k$ variables is
$$\mathfrak{E}_k(x;y) = E_{k}\left(\frac{x}{1 + x}; y_1\right)E_{k}(x; 1)$$
where $E_k$ is the ordinary top degree factorial elementary symmetric polynomial. Thus, for a dominant permutation, applying such a product of operators gives
$$\mathfrak{K}^{i + p}\cdots \mathfrak{K}^{i + 1}\mathfrak{K}^i\prod_{j = 1}\mathfrak{E}_{\lambda_j}(x;y_j)$$ 
$$=\partial^{i + p}\cdots \partial^{i + 1}\partial^iE_p(\tau^i y;1)\prod_{j=1}E_{\lambda_j}\left(\frac{x}{1 + x}; y_j\right)E_{\lambda_j}(x; 1)$$
Applying the Schubert formulas and setting $y_i=0$ yields a sum of products of ordinary elementary symmetric stable Grothendieck polynomials with the same length as $\lambda$ with the $i$th factor having $\lambda_i$ $x$ variables.
