\section{WC graphs}

\begin{definition}
A \emph{WC graph} is simply a finite subset $W\subseteq \mathbb{N}\times \mathbb{N}$ with no additional restrictions. We impose a total ordering on these pairs by declaring that $(a, b) \leq (c, d)$ if $a < c$ or $a = c$ and $b \geq d$. We define the \emph{weight} of a WC graph $W$ to be the sequence 
$$\wtt(W) = (i_1,\ldots,i_n)$$ 
where $i_j$ is the number of elements of $W$ in row $j$. We define the \emph{permutation} of a WC graph $W$ to be the permutation
$$\wof{W} = s(w_1)\bullet \cdots \bullet s(w_n)$$ 
where
$$w_i = (a_i, b_i)$$
is the $i$th element in the imposed ordering, and
$$s(a_i,b_i) = s_{a_i + b_i - 1}$$
with $\bullet$ being the Hecke product. The \emph{word} of $W$ is
$$(s(w_1),\ldots,s(w_n))$$

Define $\WC(w)$ to be the set of all WC graphs $W$ such that $\wof{W} = w$.
\end{definition}

The following is well-known.
\begin{theorem}
We have
$$\mathfrak{G}^{(\beta)}_w(x) = \sum_{W\in \WC(w)} \beta^{|W|-\ell(w)}x^{\wtt(W)}$$
\end{theorem}


\section{Grove polynomials and WC graphs}

\begin{definition}
A labeled indexed forest $(F,P)$ is an indexed forest $F$ with a function (labeling) $P:\mathrm{IN}(F)\to\mathbb{N}$ satisfying the following properties:
\begin{enumerate}
\item $P(\mathrm{left}(u)) < P(u)$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\leq P(\mathrm{right}(u))$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\geq \rho_F(u)$ for all $u\in \mathrm{IN}(F)$.
\item $P(u)\leq \lambda_F(u)$ for all $u\in \mathrm{IN}(F)$.
\end{enumerate}

The set $\mathrm{Komp}_P(F)$ is the set of functions $\kappa:\mathrm{IN}(F)\to 2^{\mathbb{N}}$ such that 

\begin{enumerate}
\item $\max(\kappa(u))\leq P(u)$ for all $u\in \mathrm{IN}(F)$.
\item $\max(\kappa(u))\leq \min(\kappa(\mathrm{left}(u)))$ for all $u\in \mathrm{IN}(F)$.
\item $\max(\kappa(u))< \min(\kappa(\mathrm{right}(u)))$ for all $u\in \mathrm{IN}(F)$.
\end{enumerate}
We have that
$$\wtt_i(\kappa) = \#\{u\in \mathrm{in}(F)\mid i\in \kappa(u)\}$$
\end{definition}


\begin{theorem}
Suppose a labeled indexed forest $(F,P)$ is such that there is a $\kappa\in\mathrm{Komp}_P(F)$ with 
$$\wtt(\kappa) = \code(F).$$ 
Then
$$\grove_F(x) = \sum_{\kappa\in\mathrm{Komp}_P(F)} \beta^{|\kappa|-|F|}x^{\wtt(\kappa)}$$
\end{theorem}



\begin{definition}
Let $W\in \WC(w)$ be a WC graph for $w$. Define via the Nadeau-Tewari insertion algorithm

and let
$$F(W) = \mathrm{shape}(P(W))$$

\end{definition}
